    <script>
        /* --- NAVIGATION LOGIC --- */
        function showPage(pageId) {
            document.querySelectorAll('.page-section').forEach(p => p.classList.remove('active'));
            document.querySelectorAll('.nav-links span').forEach(l => l.classList.remove('active'));
            
            document.getElementById(pageId).classList.add('active');
            const navLink = document.querySelector(`.nav-links span[data-target="${pageId}"]`);
            if(navLink) navLink.classList.add('active');
            window.scrollTo(0,0);
        }

        /* --- 1. QR GENERATOR --- */
        let qrCodeObj = null;
        function generateQR() {
            const text = document.getElementById("qrText").value;
            const container = document.getElementById("qrcode");
            if (!text) return alert("Please enter text or a URL");
            
            container.innerHTML = ""; 
            qrCodeObj = new QRCode(container, { text: text, width: 180, height: 180, colorDark : "#000", colorLight : "#fff" });
            setTimeout(() => { document.getElementById("downloadQrBtn").style.display = "block"; }, 300);
        }
        function downloadQR() {
            const img = document.querySelector('#qrcode img') || document.querySelector('#qrcode canvas');
            if(img && (img.src || img.toDataURL)){
                const link = document.createElement('a');
                link.href = img.src || img.toDataURL("image/png"); 
                link.download = 'qrcode.png'; link.click();
            }
        }

        /* --- 2. QR READER --- */
        const qrFileInput = document.getElementById("qrFileInput");
        const qrCanvasElem = document.getElementById("qrCanvas");
        const qrCtx = qrCanvasElem.getContext("2d", { willReadFrequently: true });
        
        qrFileInput.addEventListener("change", (e) => {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (event) => {
                const img = new Image();
                img.onload = () => {
                    qrCanvasElem.style.display = "block";
                    qrCanvasElem.width = img.width; qrCanvasElem.height = img.height;
                    qrCtx.drawImage(img, 0, 0, img.width, img.height);
                    
                    const imgData = qrCtx.getImageData(0, 0, qrCanvasElem.width, qrCanvasElem.height);
                    const code = jsQR(imgData.data, imgData.width, imgData.height, { inversionAttempts: "dontInvert" });

                    const resBox = document.getElementById("qrReaderResult");
                    resBox.style.display = "block";
                    const output = document.getElementById("qrOutputData");
                    
                    if (code) {
                        output.innerHTML = code.data.startsWith("http") ? `<a href="${code.data}" target="_blank" style="color:var(--primary);">${code.data}</a>` : code.data;
                        resBox.style.background = "#d4edda";
                    } else {
                        output.innerText = "No QR code found.";
                        resBox.style.background = "#f8d7da";
                    }
                };
                img.src = event.target.result;
            };
            reader.readAsDataURL(file);
        });

        /* --- 3. IMAGE TOOLS --- */
        const imgInput = document.getElementById('imgInput'), imgFormat = document.getElementById('imgFormat');
        const imgQuality = document.getElementById('imgQuality'), imgQualityVal = document.getElementById('imgQualityVal');
        const imgPreview = document.getElementById('imgPreview'), imgProcessBtn = document.getElementById('imgProcessBtn');
        let srcImgFile = null;

        imgQuality.addEventListener('input', e => imgQualityVal.innerText = e.target.value + '%');
        imgFormat.addEventListener('change', e => document.getElementById('imgQualityGroup').style.display = e.target.value === 'image/png' ? 'none' : 'block');
        
        imgInput.addEventListener('change', e => {
            srcImgFile = e.target.files[0];
            if (srcImgFile) {
                imgPreview.src = URL.createObjectURL(srcImgFile);
                imgPreview.style.display = 'block';
                imgProcessBtn.disabled = false;
                document.getElementById('imgStats').innerText = `Original Size: ${(srcImgFile.size/1024).toFixed(2)} KB`;
            }
        });

        imgProcessBtn.addEventListener('click', () => {
            if (!srcImgFile) return;
            const cvs = document.createElement('canvas'), ctx = cvs.getContext('2d');
            cvs.width = imgPreview.naturalWidth; cvs.height = imgPreview.naturalHeight;
            ctx.drawImage(imgPreview, 0, 0);
            
            const fmt = imgFormat.value, qual = fmt === 'image/png' ? undefined : imgQuality.value / 100;
            const dataUrl = cvs.toDataURL(fmt, qual);
            
            const link = document.createElement('a');
            link.download = `converted.${fmt.split('/')[1]}`;
            link.href = dataUrl; link.click();
            
            const estKb = (atob(dataUrl.split(',')[1]).length / 1024).toFixed(2);
            document.getElementById('imgStats').innerText += ` | New Est. Size: ${estKb} KB`;
        });

        /* --- 4. PDF TOOLS --- */
        const pdfInput = document.getElementById('pdfInput'), pdfBtn = document.getElementById('pdfCompressBtn'), pdfStat = document.getElementById('pdfStatus');
        let pdfFileObj = null;

        pdfInput.addEventListener('change', e => {
            pdfFileObj = e.target.files[0];
            if (pdfFileObj && pdfFileObj.type === "application/pdf") {
                pdfBtn.disabled = false; pdfStat.style.display = 'block';
                pdfStat.innerHTML = `Original: ${(pdfFileObj.size/1024/1024).toFixed(2)} MB`;
            }
        });

        pdfBtn.addEventListener('click', () => {
            if (!pdfFileObj) return;
            pdfBtn.disabled = true; pdfBtn.innerText = "Processing...";
            
            setTimeout(() => {
                const reader = new FileReader();
                reader.onload = e => {
                    const blob = new Blob([e.target.result], { type: 'application/pdf' });
                    const a = document.createElement('a');
                    a.href = URL.createObjectURL(blob); a.download = `optimized_${pdfFileObj.name}`; a.click();
                    
                    pdfStat.innerHTML += `<br><span style="color: green;">Done! Downloading...</span>`;
                    pdfBtn.innerText = "Compress PDF"; pdfBtn.disabled = false;
                };
                reader.readAsArrayBuffer(pdfFileObj);
            }, 1200);
        });

        /* --- 5. BMI CALCULATOR --- */
        function calcBMI() {
            const w = parseFloat(document.getElementById('bmiWeight').value);
            const h = parseFloat(document.getElementById('bmiHeight').value);
            const res = document.getElementById('bmiResult');

            if (!w || !h || w <= 0 || h <= 0) return alert("Enter valid numbers.");

            const bmi = (w / Math.pow(h/100, 2)).toFixed(1);
            let cat = "", color = "";

            if (bmi < 18.5) { cat = "Underweight"; color = "#ffc107"; }
            else if (bmi <= 24.9) { cat = "Normal weight"; color = "#28a745"; }
            else if (bmi <= 29.9) { cat = "Overweight"; color = "#fd7e14"; }
            else { cat = "Obese"; color = "#dc3545"; }

            res.style.display = "block"; res.style.backgroundColor = color; res.style.color = "#fff";
            res.innerHTML = `BMI: ${bmi} <br> <span style="font-size:0.9rem;">Category: ${cat}</span>`;
        }

        /* --- 6. GPA CALCULATOR --- */
        const gpaContainer = document.getElementById('gpaContainer');
        function addGpaSubject() {
            const div = document.createElement('div');
            div.className = 'gpa-row';
            div.innerHTML = `
                <input type="number" placeholder="Credits" class="gpa-cred" min="1">
                <select class="gpa-grade">
                    <option value="4.0">A / A+ (4.0)</option><option value="3.7">A- (3.7)</option>
                    <option value="3.3">B+ (3.3)</option><option value="3.0">B (3.0)</option>
                    <option value="2.7">B- (2.7)</option><option value="2.3">C+ (2.3)</option>
                    <option value="2.0">C (2.0)</option><option value="1.7">C- (1.7)</option>
                    <option value="1.0">D (1.0)</option><option value="0.0">F (0.0)</option>
                </select>
                <button onclick="this.parentElement.remove()" class="danger" style="margin:0;">X</button>
            `;
            gpaContainer.appendChild(div);
        }
        function calcGPA() {
            let tc = 0, tp = 0;
            document.querySelectorAll('.gpa-row').forEach(row => {
                const c = parseFloat(row.querySelector('.gpa-cred').value);
                const g = parseFloat(row.querySelector('.gpa-grade').value);
                if (c > 0) { tc += c; tp += (c * g); }
            });
            document.getElementById('gpaResult').innerText = tc === 0 ? "Add valid credits!" : `GPA: ${(tp/tc).toFixed(2)}`;
        }
        for(let i=0; i<3; i++) addGpaSubject(); // Add 3 rows by default

        /* --- 7. TEXT TOOLS (NEW) --- */
        function updateTextStats() {
            const txt = document.getElementById('textInput').value;
            document.getElementById('charCount').innerText = txt.length;
            const words = txt.trim() === '' ? 0 : txt.trim().split(/\s+/).length;
            document.getElementById('wordCount').innerText = words;
        }
        function convertCase(type) {
            const txtArea = document.getElementById('textInput');
            if(type === 'upper') txtArea.value = txtArea.value.toUpperCase();
            if(type === 'lower') txtArea.value = txtArea.value.toLowerCase();
            if(type === 'clear') txtArea.value = '';
            updateTextStats();
        }

        /* --- 8. UTILITIES (NEW) --- */
        function generatePassword() {
            const len = document.getElementById('passLength').value;
            const useUp = document.getElementById('passUpper').checked;
            const useNum = document.getElementById('passNum').checked;
            const useSym = document.getElementById('passSym').checked;
            
            let chars = "abcdefghijklmnopqrstuvwxyz";
            if (useUp) chars += "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
            if (useNum) chars += "0123456789";
            if (useSym) chars += "!@#$%^&*()_+~|}{[]:;?><,./-=";
            
            let pass = "";
            for (let i = 0; i < len; i++) pass += chars.charAt(Math.floor(Math.random() * chars.length));
            
            document.getElementById('passResult').innerText = pass;
        }
        // Generate a default password on load
        generatePassword();

        function calculateAge() {
            const dobInput = document.getElementById('dobInput').value;
            const res = document.getElementById('ageResult');
            if(!dobInput) return alert("Please select your date of birth.");
            
            const dob = new Date(dobInput);
            const today = new Date();
            
            let age = today.getFullYear() - dob.getFullYear();
            const m = today.getMonth() - dob.getMonth();
            if (m < 0 || (m === 0 && today.getDate() < dob.getDate())) {
                age--;
            }
            
            res.style.display = "block";
            res.innerHTML = `You are <strong style="color:var(--primary); font-size:1.5rem;">${age}</strong> years old.`;
        }
